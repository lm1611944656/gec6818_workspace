# 编译设备树源文件(宿主机)
dtc -I dts -O dtb -o gec6818.dtb gec6818.dts

# 在开发板的操作系统上执行
# 将开发板上的eMMC具体某一个分区挂载到开发板的文件系统中(这个才可以访问eMMC中的内容)
mount -t ext4 /dev/mmcblk0p1 /mnt

# 拷贝设备树的二进制文件到eMMC的分区1
tftp -g -r gec6818.dtb 192.168.31.62

# 重启开发板，查看内核有没有将设备树文件中定义的设备树节点生成内核中的设备节点
# 查看设备节点和设备号
cat /proc/devices

# 查看设备节点
ls -alh /sys/devices
